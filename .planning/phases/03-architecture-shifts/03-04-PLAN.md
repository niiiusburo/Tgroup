       ```typescript
           setDotkhamsLoading(true);
           try {
             const res = await fetchDotKhams({ partnerId: customerId, limit: 100 });
             if (cancelled) return;
             setDotkhams(res.items.map(dk => ({
               id: dk.id,
               name: dk.name || 'Unknown',
               totalAmount: parseFloat(dk.totalamount ?? '0'),
               residual: parseFloat(dk.amountresidual ?? '0'),
               date: dk.date ?? '',
             })));
           } catch (e) { console.error('Failed to load dotkhams', e); }
           finally { if (!cancelled) setDotkhamsLoading(false); }
         }
         loadDotKhams();
         return () => { cancelled = true; };
       }, [customerId]);
       ```
